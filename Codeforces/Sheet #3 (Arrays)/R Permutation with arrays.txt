#include <iostream>
#include <algorithm>

using namespace std;

int main()
{

int n;
cin>>n;
int arr[n];
int arr2[n];


 for (int i=0; i<n; i++){
    cin>>arr[i];
 }

 for (int j=0; j<n; j++){
    cin>>arr2[j];
 }


sort(arr, arr + n);
sort(arr2, arr2 + n);

bool isPermutation = true;



  for (int i = 0; i < n; i++) {
    if (arr[i] != arr2[i]) {
        isPermutation = false;
          break;
        }
    }





if (isPermutation){
    cout<<"yes"<<endl;
}
else{
    cout<<"no"<<endl;
}




    return 0;
}
