#include <iostream>

using namespace std;

int main()
{

int n,m;
cin>>n>>m;

long long arr[n][m];

  for (int i=0; i<n; i++){

    for (int j=0; j<m; j++){

        cin>>arr[i][j];


    }
  }

long long num;
cin>>num;
bool isFound=false;


  for (int i=0; i<n; i++){


    for (int j=0; j<m; j++){

        if (arr[i][j]==num){
            isFound=true;
            break;
        }

    }


  }

if (isFound){
    cout<<"will not take number"<<endl;
}
else {
    cout<<"will take number"<<endl;
}




    return 0;
}
