#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main()
{

long long n;
cin>>n;
long long arr[n];

 for (long long i=1; i<=n; i++){

   cin>>arr[i];
 }

long long minimum=arr[1];



  for (long long j=2; j<=n; j++){


    if (arr[j]<minimum){
        minimum=arr[j];
    }


  }

 for (int q=1; q<=n; q++){

    if (minimum==arr[q]){
        cout<<minimum<<" "<<q<<endl;
        break;

    }
 }


    return 0;
}
