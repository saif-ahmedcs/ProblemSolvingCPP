#include <iostream>
#include <cctype>
#include <cstring>

using namespace std;

int main()
{

int a,b;
cin>>a>>b;
int x=a+b+1;

string s;
cin>>s;

bool isYes=true;

 for (int i=0; i<x; i++){

    if (i==a){
        if (s[i]!='-'){
           isYes=false;
        }
        continue;
    }

    else if (!isdigit(s[i])){
        isYes=false;
    }


 }


 if (isYes && x==s.length()){
    cout<<"Yes"<<endl;
 }
 else {
    cout<<"No"<<endl;
 }



    return 0;
}
