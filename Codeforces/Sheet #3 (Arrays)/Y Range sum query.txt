#include <iostream>

using namespace std;

int main()
{


long long n, q;
cin>>n>>q;

long long arr[n];
long long PreSum[n];



for (int i=1; i<=n; i++){
    cin>>arr[i];
     if (i == 1){
        PreSum[i] = arr[i];
        }
      else {
    PreSum[i] = arr[i] + PreSum[i-1];
}
}

long long sum=0;


  while (q--){

    long long l,r;
    cin>>l>>r;

    if (l==1){
        sum = PreSum[r];
    }

    else {

     sum = PreSum[r] - PreSum[l-1];

  }

  cout<<sum<<endl;

  }


    return 0;
}
