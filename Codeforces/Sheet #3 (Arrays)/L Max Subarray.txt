#include <iostream>
#include <vector>
using namespace std;

int main() {


int t;
cin>>t;

vector <long long> vec(100);


 while (t--){

  int n;
  cin>>n;

    for (int i=0; i<n; i++){
        cin>>vec[i];
    }



    for (int i =0; i<n; i++){

     int maximum = vec[i];

        for (int j=i; j<n; j++){


            for (int z=i; z<=j; z++){

                if (vec[z]>maximum){
                    maximum=vec[z];
                }
            }

            cout<<maximum<<" ";
        }



    }

    cout<<endl;




 }
    return 0;
}
