#include <iostream>

using namespace std;

int main()
{


long long t;
cin>>t;

while (t--){

    int n;
    cin>>n;

   long long arr[n];


    for (long long i = 1; i <= n; i++) {
            cin >> arr[i];
    }



long long Minsum = arr[1] + arr[2] + 2 - 1;
long long sum=0;



  for (long long i = 1; i<=n-1; i++){


    for (long long j=i+1; j<=n; j++){

        sum = arr[i]+arr[j]+j-i;

        if (sum<Minsum){
            Minsum = sum;
        }

    }

  }

 cout<<Minsum<<endl;

}



    return 0;
}
