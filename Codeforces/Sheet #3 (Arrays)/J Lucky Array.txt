#include <iostream>

using namespace std;

int main()
{

long long n;
cin>>n;
long long arr[n];


  for (long long i=0; i<n; i++){

     cin>>arr[i];

  }

long long minumum = arr[0];
long long occurrence = 0;

for (long long j=0; j<n; j++){

    if (arr[j]<minumum){
        minumum = arr[j];
    }

}


for (long long q = 0; q<n; q++){

    if (arr[q]==minumum){

     occurrence++;

    }

}


if (occurrence %2 != 0){
    cout<<"Lucky"<<endl;
}
else {
    cout<<"Unlucky"<<endl;
}



    return 0;
}
