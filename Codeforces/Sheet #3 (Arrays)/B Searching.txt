#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main()
{

long long n;
cin>>n;
long long arr[n];
bool isFound = false;

  for (long long i=0; i<n; i++){
    cin>>arr[i];

  }



int number;
cin>>number;


  for (long long j=0; j<n; j++){

     if (arr[j]==number){
        cout<<j<<endl;
        isFound = true;
        break;
     }


  }


  if (!isFound){
    cout <<-1;
  }



    return 0;
}
