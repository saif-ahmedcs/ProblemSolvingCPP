#include <iostream>

using namespace std;

int main()
{


int n;
cin>>n;
long long arr[n][n];

  for (long long i=0; i<n; i++){

    for (long long j=0; j<n; j++){

        cin>>arr[i][j];

    }

  }


long long sum1=arr[0][0], sum2=arr[0][n-1];


  for (long long i=1; i<n; i++){


      sum1 += arr[i][i];

      sum2 += arr[i][n-1-i];

  }


long long result = sum1-sum2;

cout<<abs(result)<<endl;



    return 0;
}
