#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main()
{

long long n;
cin>>n;
int arr[n];

 for (long long i=0; i<n; i++){
    cin>>arr[i];

     if (arr[i]>0){
        arr[i]=1;
     }
     else if (arr[i]==0){
        arr[i]=0;
     }
     else {
        arr[i]=2;
     }
 }


 for (long long j=0; j<n; j++){
    cout<<arr[j]<<" ";
 }




    return 0;
}
