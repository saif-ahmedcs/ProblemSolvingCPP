#include <iostream>
#include <string>
#include <cstring>

using namespace std;

int main()
{

string s;
getline(cin,s);
int x = 0;


 for (int i = 0; i<s.length(); i++){

    if (s[i]== ' ' && x == 0){

        for (int j = i-1; j>=0; j--){
            cout << s[j] ;
        }
        cout << " ";
     if (i!=0){
        x = i;
     }
     else {
        x = 1;
     }
    }

    else if (s[i] == ' ' && x != 0){

        for (int q = i-1; q>x; q--){
            cout << s[q] ;
        }
        cout << " ";

        x = i;
    }

    else if (i == s.length()-1){

        for (int w = i; w>x; w--){
            cout << s[w] ;
        }
    }



 }


    return 0;
}
