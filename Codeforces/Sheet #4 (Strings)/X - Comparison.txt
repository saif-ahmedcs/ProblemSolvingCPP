#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{

string s,s1,s2,s3;
cin >> s;
string smallest = "";


   if (s.length() == 1) { 
        cout << s << endl;
        return 0;
    }


   for (int i=0; i<s.length()-1; i++){

     s1.clear();
     s2.clear();

    for (int j=0; j<=i; j++){

        s1.push_back(s[j]);

    }

     for (int q=i+1; q<s.length(); q++){
        s2.push_back(s[q]);

     }

    sort (s1.begin(),s1.end());
    sort (s2.begin(),s2.end());

    s3 = s1+s2;
    if (smallest.empty()){
    smallest = s3;
    }
    else if (smallest > s3){
        smallest = s3;
    }
   }


  cout <<smallest;





    return 0;
}
