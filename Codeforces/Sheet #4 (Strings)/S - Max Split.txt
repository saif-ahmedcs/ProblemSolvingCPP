#include <iostream>
#include <string>

using namespace std;

int main()
{

string s;
cin>>s;
int numofL = 0;
int numofR = 0;
int counter = 0;
string str = "";



 for (int i = 0; i<s.length(); i++){

        if (s[i]=='L'){
          numofL ++;
        }
        else if (s[i]=='R'){
          numofR ++;
        }

        if (numofL == numofR){
            counter ++;
        }

 }

cout << counter << endl;


 for (int i=0; i<s.length(); i++){


        if (s[i]=='L'){
          str.push_back(s[i]);
          numofL ++;
        }
        else if (s[i]=='R'){
          str.push_back(s[i]);
          numofR ++;
        }

        if (numofL == numofR){
          cout << str << endl;
          str.clear();
        }
 }



    return 0;
}
