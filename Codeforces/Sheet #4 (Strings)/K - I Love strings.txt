#include <iostream>
#include <string>
using namespace std;

int main() {

int n;
string s1,s2;
cin>>n;

while (n--){

cin>>s1>>s2;

if (s1.length()>= s2.length()){

    for (int i=0; i<s1.length(); i++){

        cout<<s1[i];

        if (i<=s2.length()-1){
            cout<<s2[i];
        }

    }

}

else {

    for (int i=0; i<s2.length(); i++){


        if (i<=s1.length()-1){
           cout<<s1[i];
        }


        cout<<s2[i];
    }


    }


cout<<endl;

}


    return 0;
}
