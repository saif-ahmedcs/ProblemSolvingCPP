#include <iostream>
#include <string>
using namespace std;

int main() {


string s;
cin>>s;

bool ispalindrome = true;
int length = s.length();


  for (int i=0; i<length/2; i++){

    if (s[i] != s[length - 1 - i]){
        ispalindrome = false;
        break;
    }
  }


if (ispalindrome){
    cout<<"YES"<<endl;
}
else {
    cout<<"NO"<<endl;
}


    return 0;
}
