#include <iostream>

using namespace std;

int main()
{

int t;
cin >> t;

 while (t--){

    string s;
    cin >> s;
    bool isGood = false;

    if (s.length() < 3){
        cout<<"Bad"<<endl;
    }


    else {

    for (int i=0; i<=s.length()-2; i++){

        if (s[i]=='1' && s[i+1]== '0' && s[i+2]=='1'){
            isGood = true;
            break;
        }

        else if (s[i]=='0' && s[i+1]=='1' && s[i+2]=='0'){
            isGood = true;
            break;
        }
    }

    if (isGood){
        cout<<"Good"<<endl;
    }
    else {
        cout<<"Bad"<<endl;
    }
    }
 }



    return 0;
}
