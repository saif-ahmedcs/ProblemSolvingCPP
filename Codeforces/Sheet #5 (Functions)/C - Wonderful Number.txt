#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;


int IsOdd (int n){

  if (n % 2 == 0){
    return 0;
  }

   else {
    return 1;
}
}

vector <int> bits = {};

void BitsArePalindrome (int n){

  while (n!=0){
    bits.push_back(n%2);
    n/=2;
  }

  reverse(bits.begin(),bits.end());

  bool isPalindrome = true;

  for (int i=0; i<bits.size()/2; i++){

    if (bits[i] != bits[bits.size() - 1 - i]){
        isPalindrome = false;
        break;
    }
  }

  if (isPalindrome){
    cout<<"YES"<<endl;
  }
  else {
    cout<<"NO"<<endl;
  }

}

int main()
{

int n;
cin >> n;

 if (IsOdd(n)){
    BitsArePalindrome(n);
 }
 else {
    cout <<"NO"<<endl;
 }


    return 0;
}
