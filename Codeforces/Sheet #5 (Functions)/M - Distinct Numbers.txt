#include <iostream>

using namespace std;

int FindDistnictNums (int length , int arr[]){

int freqArr[2001]={0};
int counter = 0;


 for (int i=0; i<length; i++){

     if(freqArr[arr[i]] == 0){
       freqArr[arr[i]]++;
       counter ++;
 }

 }

   return counter;

}

int main()
{


int length;
cin >>length;

int arr[length];

 for (int i=0; i<length; i++){
    cin>>arr[i];
 }

 cout << FindDistnictNums(length,arr)<<endl;

    return 0;
}
