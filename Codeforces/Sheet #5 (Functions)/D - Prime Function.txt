#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

void IsPrimeOrNot(int t){

int n;

while (t--){

 cin >> n;
 if (n <= 1){
    cout<<"NO"<<endl;
    continue;
 }


 bool isprime = true;


 for (int i=2; i*i<=n; i++){

    if (n % i == 0){
        isprime = false;
        break;
    }
 }

  if (isprime){
    cout << "YES"<<endl;
  }
  else {
    cout <<"NO"<<endl;
  }

}

}

int main()
{


int t;
cin >>t;

IsPrimeOrNot(t);


    return 0;
}
